int ValueOfAudioPin = 11;
int intialSensorValue=0;
int LightLedValue = 13;
void loop()
{
  tone(ValueOfAudioPin, 500);
  intialSensorValue = digitalRead(ValueOfAudioPin);
  Serial.println(intialSensorValue);
  digitalWrite(LightLedValue, intialSensorValue);
  delay(200);
}

void setup()
{
  pinMode(ValueOfAudioPin, OUTPUT);
  pinMode(LightLedValue, OUTPUT);
  Serial.begin(9600);
}
