#include <Servo.h>

int valueOfLed=6;
Servo newServo;
int sensorPirValue=2;
int intialSensorValue = 0;


void loop()
{
  int sensorValue = digitalRead(sensorPirValue);
  Serial.println(sensorValue);
  if(sensorValue==HIGH){
    digitalWrite(valueOfLed,HIGH);
    newServo.write(70);
  }
  else{
    digitalWrite(valueOfLed,LOW);
    newServo.write(10);
  }
  intialSensorValue = analogRead(A0);
  Serial.println(intialSensorValue);
  int fade = map(intialSensorValue, 54, 974, 255, 0);
  Serial.println(fade);
  analogWrite(13, fade);
  delay(100);
}

void setup()
{
  pinMode(sensorPirValue,INPUT);
  pinMode(valueOfLed,OUTPUT);
  newServo.attach(9);
   // pin for ldr
  pinMode(A0, INPUT);
  // pin for LED
  pinMode(13, OUTPUT);
  // starts the serial monitor
  Serial.begin(9600);
}
