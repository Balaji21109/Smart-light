#include <Servo.h>
Servo myservo;
int led=6;
int pir_sensor=2;

void setup()
{
  pinMode(pir_sensor,INPUT);
  pinMode(led,OUTPUT);
  myservo.attach(9);
  Serial.begin(9600);

}

void loop()
{
  int value = digitalRead(pir_sensor);
  Serial.println(value);
  if(value==HIGH){
    digitalWrite(led,HIGH);
    myservo.write(70);
  }
  else{
    digitalWrite(led,LOW);
    myservo.write(10);
  }
  
  delay(10);
  

}
